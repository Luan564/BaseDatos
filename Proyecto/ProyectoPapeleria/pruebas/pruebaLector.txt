    id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
    codigo_barras VARCHAR(45) UNIQUE NOT NULL,
    nombre VARCHAR(50) NOT NULL UNIQUE,
    cantidad_piezas INT UNSIGNED DEFAULT 0,
    precio DECIMAL(4,2)
);
